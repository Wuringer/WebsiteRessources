void reconnect() {
  // Loop until we're reconnected
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    // Attempt to connect
    if (client.connect("ESP32Client",mqttuser,mqttpassword)) {
      Serial.println("connected");
      // Subscribe
      //client.subscribe("test/vitalge01");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      // Wait 5 seconds before retrying
      delay(5000);
    }
  }
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop(); 
  if ((millis() - timing) > 5000)
  {
    timing = millis();

    if (iteration > 4){iteration = 0;}
    pumpcontrol(iteration);
    Serial.print("Iteration: "); Serial.println(iteration);
    iteration+=1;

    if(iteration == 3){digitalWrite(LUFTPUMPE,HIGH);}else{digitalWrite(LUFTPUMPE,LOW);};
    
    
    sensors.requestTemperatures(); 
    temp = (float)sensors.getTempCByIndex(0); //temp = 100.00;
    OD = (float)analogRead(ODPROBE); //OD = 50.00;
    pH = (float)analogRead(PHPROBE); //pH = 1024.00;
    
    
    // Convert the value to a char array
    char tempString[8];
    dtostrf(temp, 1, 2, tempString);
    Serial.print("Temperature: ");
    Serial.println(tempString);
    
    char ODString[8];
    dtostrf(OD, 1, 2, ODString);
    Serial.print("Humidity: ");
    Serial.println(ODString);

    char pHString[8];
    dtostrf(pH, 1, 2, pHString);
    Serial.print("pH: ");
    Serial.println(pHString);
    
    if(client.connected())
    {
      client.publish("vitalge/c", tempString);
      client.publish("vitalge/od", ODString);
      client.publish("vitalge/ph", pHString);
    }
  }
}

void pumpcontrol(unsigned int state)
{
  switch(state)
  {
    case 0: digitalWrite(ENA1,LOW);digitalWrite(ENA2,LOW); Serial.println("Off");
    break;
    case 1: digitalWrite(ENA1,HIGH);digitalWrite(ENA2,LOW); Serial.println("Left");
    break;
    case 2: digitalWrite(ENA1,LOW);digitalWrite(ENA2,LOW); Serial.println("Off");
    break;
    case 3: digitalWrite(ENA1,LOW);digitalWrite(ENA2,HIGH); Serial.println("Right");
    break;
    default: digitalWrite(ENA1,LOW);digitalWrite(ENA2,LOW); Serial.println("Off");
    break;
  }
}
